CREATE CORTEX SEARCH SERVICE BANKING.VARO_FEEDBACK_SEARCH
       ON FEEDBACK_TEXT
       ATTRIBUTES CUSTOMER_ID, SENTIMENT
       WAREHOUSE = <your_warehouse>  -- e.g., COMPUTE_WH
       TARGET_LAG = '1 hour'
       AS
       SELECT FEEDBACK_ID, CUSTOMER_ID, FEEDBACK_TEXT, FEEDBACK_DATE, SENTIMENT
       FROM BANKING.CUSTOMER_FEEDBACK;
